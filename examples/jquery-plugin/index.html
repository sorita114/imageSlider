1. Defaults Example

2. Change SlideTime Example

});

3. Disabled button

            $(".s3").imageSlider({
 usedButton : true,
 usedButtonNext : false,
 usedButtonPrev : false,
 usedButtonPause : true,
 usedButtonResume : true
});

4. Auto Slider False

            $(".s4").imageSlider({

});

5. isLoop False

            $(".s5").imageSlider({
 isLoop : false
});

6. Default Theme

            $(".s6").imageSlider({
